+def check_method_for_static_file(request, dispatch_result):
+    """Return 405 for unsafe methods on static files.
+
+    Any request for a static file with a method other than GET or HEAD
+    should be rejected with 405 (Method Not Allowed).
+
+    According to RFC 7231, safe methods are: GET, HEAD, OPTIONS, TRACE
+    """
+    # Check if this is a static file (not a .spt file and not dynamic)
+    if dispatch_result.match and not dispatch_result.match.endswith('.spt'):
+        # Static files should only be served with safe methods
+        safe_methods = {b'GET', b'HEAD', b'OPTIONS', b'TRACE'}
+        if request.line.method not in safe_methods:
+            raise Response(405)
